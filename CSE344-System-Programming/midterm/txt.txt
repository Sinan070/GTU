    do (position-if-not delimiterp line :start (1+ end))
    :for end = (and beg (position-if delimiterp line :start beg))
    :when beg :collect (subseq line beg end)
    :while end)



(defun my-split (line list (delimiterp #'delimiterp))
  (loop :for beg = (position-if-not delimiterp line)
    do (position-if-not delimiterp line :start (1+ end))
    :for end = (and beg (position-if delimiterp line :start beg))
    :when beg :collect (subseq line beg end)
    :while end)
